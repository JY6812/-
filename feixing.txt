local function LMCN_fake_script() -- famuwanjiagongneng.LocalScript 
	local script = Instance.new('LocalScript', famuwanjiagongneng)

	local function DIHTFDD_fake_script() 
		local script = Instance.new('LocalScript', famuwanjiagongneng)
		local isDragging = false
		local startPosition
		local startMousePosition
	
		famuwanjiagongneng.InputBegan:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
				isDragging = true
				startPosition = famuwanjiagongneng.Position
				startMousePosition = input.Position
			end
		end)
	
		famuwanjiagongneng.InputChanged:Connect(function(input)
			if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and isDragging then
				local delta = input.Position - startMousePosition
				local newPosition = UDim2.new(
					startPosition.X.Scale,
					startPosition.X.Offset + delta.X,
					startPosition.Y.Scale,
					startPosition.Y.Offset + delta.Y
				)
				famuwanjiagongneng.Position = newPosition
			end
		end)
	
		famuwanjiagongneng.InputEnded:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
				isDragging = false
			end
		end)
	end
	coroutine.wrap(DIHTFDD_fake_script)()
end
